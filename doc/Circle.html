def initialize(**args)
  @x = args.fetch(:x, 0)
  @y = args.fetch(:y, 0)
  @radius = args.fetch(:radius, 10)
  @thickness = args.fetch(:thickness, 1)
  @color = args.fetch(:color, Color::BLACK)

  @changed = true
  @id = "circle:#{@@id += 1}"
end
