def to_primitives
  return unless visible?

  if frame_w && background
    [
      relative_rect.solid!(color.to_h),
      relative_rect.sprite!(path: background, source_x: frame_w * (frame - 1), source_w: frame_w)
    ] + @children.to_primitives
  else
    super
  end
end
