def initialize(**args)
  @id = @@window_id
  @x = args.fetch(:x, 0)
  @y = args.fetch(:y, 0)
  @w = args.fetch(:w, 0)
  @h = args.fetch(:h, 0)
  @color = args.fetch(:color, Color::LIGHT_GREY)
  @text = args.fetch(:text, "#{self.class.name}#{@@window_id += 1}")
  @visible = args.fetch(:visible, true)
  @visible = args.fetch(:visible, true)
  @background = args.fetch(:background, nil)

  @children = WindowCollection.new(self)

  super(args)
end
