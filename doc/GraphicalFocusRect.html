def to_primitives
  return unless visible?

  relative_rect = child.relative_rect

  if w.to_i > 0
    relative_rect[:x] = child.x - (w - child.w) / 2
    relative_rect[:w] = w
  end

  if h.to_i > 0
    relative_rect[:y] = child.y - (h - child.h) / 2
    relative_rect[:h] = h
  end

  relative_rect.sprite!(path: path)
end
